using System.Security.Cryptography;

namespace ConsoleApp258
{
    
    internal class Program
    {

        static int Calc(int n)
        {
            int result = 1;
            for (int i = 1; i <= n; i++) 
            {
                result *= i;
            }
            return result;
        }

        static void Main(string[] args)
        {
          
            int n = int.Parse(Console.ReadLine());
            
            Console.WriteLine(Calc(n));
        }
       
        
        
    }
}
