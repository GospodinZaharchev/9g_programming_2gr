namespace ConsoleApp263
{
     
    internal class Program
    {
        static int Calc(int n)
        {
            int result = 1;
            for (int i = 1; i <= n; i++)
            {
                result *= i;
            }
            return result ;
        }
        static void Main(string[] args)
        {
             Calc(int.Parse(Console.ReadLine()),int.Parse(Console.ReadLine())); 
        }
    }
}
