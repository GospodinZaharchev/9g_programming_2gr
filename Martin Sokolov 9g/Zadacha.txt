namespace ConsoleApp266
{
    class Count
    {
        public string a;
        public int b;
        public Count(string a, int b)
        {
            this.a = a;
            this.b = b;
        }
        public void zadacha()
        {
            for(int i = 0; i < b; i++)
            {
                Console.WriteLine(a);
            }
        }
        
    }
    internal class Program
    {
        static void Main(string[] args)
        {
           Count count1 = new Count(Console.ReadLine(), int.Parse(Console.ReadLine()));
            count1.zadacha();
        }
    }
}
